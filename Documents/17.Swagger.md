### [Swagger](https://docs.nestjs.com/openapi/introduction)


- Setup swagger
```ts
patchNestJsSwagger()
  const config = new DocumentBuilder()
    .setTitle('Ecommerce API')
    .setDescription('The API for the ecommerce application')
    .setVersion('1.0')
    .addBearerAuth()
    .addApiKey(
      {
        name: 'authorization',
        type: 'apiKey',
      },
      'payment-api-key',
    )
    .build()

  const documentFactory = () => SwaggerModule.createDocument(app, config)
  SwaggerModule.setup('api', app, documentFactory, {
    swaggerOptions: {
      persistAuthorization: true,
    },
  })

// user.controller.ts thêm decorator vào để apply authentication cho từng api.
@ApiBearerAuth()
// payment.controller.ts thêm decorator vào payment-api-key
@ApiSecurity('payment-api-key')

// product.controller.ts Thêm decorator vào để chỉ ro cho swager biết params api là productId
@ApiParam({ name: 'productId', type: String })
```
